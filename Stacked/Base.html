def association(assoc)
  instance_eval do
    assoc = assoc.to_s
    define_method("#{assoc}=") do |records|
      instance_variable_set("@#{assoc}", records.map { |record| "Stacked::#{assoc.classify}".constantize.new(record) })
    end

    define_method(assoc) { instance_variable_get("@#{assoc}") }
  end
end
